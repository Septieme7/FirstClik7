6. Base de Données Room

ArticleDao.kt 
package com.example.firstclik7
import androidx.room.*
@Dao interface ArticleDao {
    @Insert suspend fun inserer(article: Article)
    @Query("SELECT * FROM articles ORDER BY id DESC") suspend fun obtenirTous(): List<Article>
    @Delete suspend fun supprimer(article: Article)
}

10. Logique Principale

MainActivity.kt 
package com.example.firstclik7
import android.os.Bundle; import androidx.appcompat.app.AppCompatActivity; import androidx.lifecycle.lifecycleScope; import androidx.recyclerview.widget.RecyclerView
import com.google.android.flexbox.*; import kotlinx.coroutines.launch
class MainActivity : AppCompatActivity() {
    private lateinit var dao: ArticleDao; private lateinit var adapter: ArticleAdapter
    override fun onCreate(s: Bundle?) {
        super.onCreate(s); setContentView(R.layout.activity_main)
        dao = ArticleDatabase.getInstance(this).articleDao()
        val rv = findViewById<RecyclerView>(R.id.recyclerView)
        rv.layoutManager = FlexboxLayoutManager(this).apply { flexDirection = FlexDirection.ROW; justifyContent = JustifyContent.FLEX_START }
        charger()
    }
    private fun charger() { lifecycleScope.launch { val list = dao.obtenirTous(); adapter = ArticleAdapter(list, { /* zoom */ }, { /* del */ }); findViewById<RecyclerView>(R.id.recyclerView).adapter = adapter } }
}
